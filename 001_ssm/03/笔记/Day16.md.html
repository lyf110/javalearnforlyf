       switch (stockMessage.getStockHead().getRequestType()) {
           case PLACE_ORDER:
               processPlaceOrder(channelHandlerContext, stockMessage);
               break;
           case GET_ORDER:
               processGetOrder(channelHandlerContext, stockMessage);
               break;
           default:
               log.error("Process in StockDealProcessor.processRequest method => Not Match RequestType!");
       }
   }


   /**
    * 
    * @param channelHandlerContext
    * @param stockMessage
    */
   private void processPlaceOrder(ChannelHandlerContext channelHandlerContext, StockMessage stockMessage) {
       ResponseMessage responseMessage = null;
       StockHeadProto.StockHead stockHead = null;
       try {

           // 
           PlaceOrderData orderData = stockMessage.getPlaceOrderData();
           // 
           PlaceOrderRequest request = PlaceOrderRequest.newBuilder().setAccountId(getUserAccountId(channelHandlerContext))
                   .setDirection(orderData.getDirection()).setExecPrice(orderData.getExecPrice())
                   .setStockId(orderData.getStockId()).setType(orderData.getType())
                   .setExecVolume(orderData.getExecVolume()).build();
           final PlaceOrderReponse response = stockTradeDealServiceStub.placeOrder(request);
           log.info("Process in StockDealProcessor.processRequest receive: " + response);

           // 
           stockHead = generateStockHead(stockMessage.getStockHead(), response.getStatus());
           PlaceOrderResp orderResp = PlaceOrderResp.newBuilder().setDealId(response.getDealId()).build();
           responseMessage = ResponseMessage.newBuilder().setStockHead(stockHead).setMessage(response.getMessage()).setPlaceOrderResp(orderResp).build();

       }catch(ComponentException e) {
           // 
           log.error(e.getMessage(), e);
           stockHead = generateStockHead(stockMessage.getStockHead(), e.getErrorCodeEnum().getCode());
           responseMessage = ResponseMessage.newBuilder().setStockHead(stockHead).setMessage(e.getErrorCodeEnum().getMessage()).build();
       }catch (Exception e) {
           // 
           log.error(e.getMessage(), e);
           stockHead = generateStockHead(stockMessage.getStockHead(), GlobalConstants.STATUS_ERROR);
           responseMessage = ResponseMessage.newBuilder().setStockHead(stockHead).setMessage(e.getMessage()).build();
       }

       channelHandlerContext.writeAndFlush(responseMessage);
   }

       if(null != quoteLast) {
           // 
           //  =  -  * 
           long diffPrice = MathUtils.sub(quoteLast.getLastPrice(), MathUtils.round2Long(position.getCostPrice()));
           long profit = MathUtils.mul(diffPrice, position.getVolume());
           position.setProfit(profit);
           position.setLastPrice(MathUtils.div(quoteLast.getLastPrice(), 100, 2));
           //  =  /  * 
           position.setProfitRate(MathUtils.div2Long(profit, MathUtils.mul(position.getCostPrice(), position.getVolume()), 4));
       }
