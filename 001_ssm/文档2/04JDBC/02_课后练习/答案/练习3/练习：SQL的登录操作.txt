需求：
	1). 创建一个工具类JdbcUtil
	2). 用户通过控制台输入用户名和密码，实现登录的功能。如果用户名和密码输入正确，则登录成功。
	
1. 有一张用户表
	CREATE TABLE USER (
		id INT AUTO_INCREMENT PRIMARY KEY,
		NAME VARCHAR(50),
		PASSWORD VARCHAR(50)
	)

2. 添加几条用户记录
	INSERT INTO user (NAME, PASSWORD) VALUES ('admin', '123');

3.使用Statement字符串拼接的方式实现用户的登录, 用户在控制台上输入用户名和密码。

	package com.itheima.util;

	import java.sql.Connection;
	import java.sql.DriverManager;
	import java.sql.ResultSet;
	import java.sql.SQLException;
	import java.sql.Statement;

	/**
	 * 数据库访问类的工具类
	 *  1. 加载驱动
		2. 得到数据库的连接 
		3. 关闭所有打开的资源
	 * 
	 */
	public class JdbcUtil {

		private static final String URL = "jdbc:mysql://localhost:3306/day16";
		private static final String USER = "root";
		private static final String PASSWORD = "root";
		private static final String DRIVER = "com.mysql.jdbc.Driver";

		/**
		 * 类的驱动只需加载1次
		 */
		static {
			try {
				Class.forName(DRIVER);
			} catch (ClassNotFoundException e) {
				e.printStackTrace();
			}
		}

		/**
		 * 得到数据库的连接
		 */
		public static Connection getConnection() {
			try {
				return DriverManager.getConnection(URL, USER, PASSWORD);
			} catch (SQLException e) {
				e.printStackTrace();
			}
			return null;
		}

		/**
		 * 释放所有的资源
		 */
		public static void close(Connection conn, Statement stmt, ResultSet rs) {
			try {
				if (rs != null) {
					rs.close();
				}
			} catch (SQLException se) {
				se.printStackTrace();
			}
			try {
				if (stmt != null) {
					stmt.close();
				}
			} catch (SQLException se) {
				se.printStackTrace();
			}
			try {
				if (conn != null) {
					conn.close();
				}
			} catch (SQLException se) {
				se.printStackTrace();
			}
		}
		
		/**
		 * 释放资源
		 * @param conn
		 * @param stmt
		 */
		public static void close(Connection conn, Statement stmt) {
			close(conn,stmt,null);
		}
	}

	/**
		需求：用户通过控制台输入用户名和密码，实现登录的功能。
		如果用户名和密码输入正确，则登录成功。
	 */
	public class Demo1 {
		
		public static void main(String[] args) throws SQLException {
			Scanner scanner = new Scanner(System.in);
			System.out.println("请输入用户名:");
			String name = scanner.nextLine();
			System.out.println("请输入密码:");
			String password = scanner.nextLine();
			//访问数据库，实现登录的功能
			//1. 得到连接
			Connection connection = JdbcUtil.getConnection();
			//2. 创建语句对象
			Statement stmt = connection.createStatement();
			String sql = "select * from user where name='" + name + "' and password='" + password + "'";
			//3. 执行SQL
			ResultSet rs = stmt.executeQuery(sql);
			System.out.println(sql);
			//4.判断是否有结果
			if (rs.next()) {
				//表示有记录
				System.out.println("欢迎您！" + name + "，登录成功！");
			}
			else {
				System.out.println("登录失败，请重试！");
			}
			//5.关闭结果集
			JdbcUtil.close(connection, stmt, rs);
		}

	}
